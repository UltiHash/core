function(uh_add_test name)
    add_executable(${name} ${name}.cpp)
    target_link_libraries(${name} ${ARGN} ${Boost_LIBRARIES})
    add_test(NAME ${name} COMMAND $<TARGET_FILE:${name}>)
endfunction()
